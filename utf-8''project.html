The Assignment¶

Take a ZIP file) of images and process them, using a library built into python that you need to learn how to use. A ZIP file takes several different files and compresses them, thus saving space, into one single file. The files in the ZIP file we provide are newspaper images (like you saw in week 3). Your task is to write python code which allows one to search through the images looking for the occurrences of keywords and faces. E.g. if you search for "pizza" it will return a contact sheet of all of the faces which were located on the newspaper page which mentions "pizza". This will test your ability to learn a new (library), your ability to use OpenCV to detect faces, your ability to use tesseract to do optical character recognition, and your ability to use PIL to composite images together into contact sheets.

Each page of the newspapers is saved as a single PNG image in a file called images.zip. These newspapers are in english, and contain a variety of stories, advertisements and images. Note: This file is fairly large (~200 MB) and may take some time to work with, I would encourage you to use small_img.zip for testing.

import zipfile

from PIL import Image
import pytesseract
import cv2 as cv
import numpy as np

# loading the face detection classifier
face_cascade = cv.CascadeClassifier('readonly/haarcascade_frontalface_default.xml')

# the rest is up to you!

from PIL import ImageDraw

dicts = []  # a list of diction objects, each stores filename, PIL Image, bounding boxes, and text
# deal with zip file
with zipfile.ZipFile('readonly/images.zip', 'r') as testzip:
    # files = testzip.infolist()
    for i in testzip.infolist():
        dict = {}
        # print(i.filename)
        dict['name'] = i.filename
        with testzip.open(i) as file:
            pil_img = Image.open(file)
            dict['image'] = pil_img
            # pil_img.show()
            # print(pil_img.mode)
            data = testzip.read(file.name)
            cv_img = cv.imdecode(np.frombuffer(data, np.uint8), 1)

            # cv_img = cv.cvtColor(np.asarray(pil_img, np.uint8), cv.COLOR_RGB2BGR) #convert PIL Image to cv2 image
            gray = cv.cvtColor(cv_img, cv.COLOR_BGR2GRAY)
            #gray = cv.equalizeHist(gray)

            faces = face_cascade.detectMultiScale(gray, 1.3, 5)  # Scale 1.3 5
            # print(faces)
            # test faces
            dict["faces"] = faces
            img = pil_img.copy()
            drawing = ImageDraw.Draw(img)
            for x, y, w, h in faces:
                drawing.rectangle((x, y, x + w, y + h), outline="red")
            img.show()

            # OCR text
            text = pytesseract.image_to_string(pil_img.convert('1'))
            dict['text'] = text
            # print(text)

        dicts.append(dict)


def findfiles(target):
    fnames = []
    for dict in dicts:
        if target in dict['text']:
            fnames.append(dict['name'])
    return fnames


def find_faces(fname):
    images = []
    for dict in dicts:
        if dict['name'] == fname:
            for (x, y, w, h) in dict['faces']:
                img = dict['image']
                face_object = img.crop((x, y, x+w, y+h))
                #face_object.show()
                images.append(face_object)
    return images


def make_contact_sheet(images):
    first_image = images[0]
    MAX_SIZE = (100, 100)
    if len(images) % 5 == 0:
        rownum = len(images) / 5
    else:
        rownum = len(images) // 5 + 1
    contact_sheet = Image.new(first_image.mode, (MAX_SIZE[0] * 5, int(rownum * MAX_SIZE[1])))
    x = 0
    y = 0
    for img in images:
        # print(img.width, img.height)  # find the appropriate max size
        img.thumbnail(MAX_SIZE)  # resize the img
        # paste the current image into the contact sheet
        contact_sheet.paste(img, (x, y))
        # update x and y position
        if x + first_image.width == contact_sheet.width:
            x = 0
            y = y + first_image.height
        else:
            x = x + first_image.width
    # resize and return the contact sheet
    # contact_sheet = contact_sheet.resize((int(contact_sheet.width / 2), int(contact_sheet.height / 2)))
    return contact_sheet


def test(st):
    #str = "Mark"
    fnames = findfiles(st)
    for fname in fnames:
        print("Results found in file {}".format(fname))
        if len(find_faces(fname)) != 0:
            contact_sheet = make_contact_sheet(find_faces(fname))
            contact_sheet.show()
            display(contact_sheet)
        else:
            print("But there were no faces in that file!")
